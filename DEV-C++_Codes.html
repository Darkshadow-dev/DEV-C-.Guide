Basic C++ Code Examples

Here's a simple C++ code example for addition:


#include <iostream>
using namespace std;

int main() {
    int a, b, sum;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    sum = a + b;
    cout << "Sum = " << sum << endl;
    return 0;
}

Here's a simple C++ code example for multiplication:


#include <iostream>
using namespace std;

int main() {
    int a, b, product;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    product = a * b;
    cout << "Product = " << product << endl;
    return 0;
}

Here's a more complex example that calculates the square root of a number:


#include <iostream>
#include <cmath>
using namespace std;

int main() {
    double number = 9.0;
    double result = sqrt(number);
    cout << "Square root of " << number << " is " << result << endl;
    return 0;
}

Loop examples:

For Loop:


#include <iostream>
using namespace std;

int main() {
    for (int i = 0; i < 6; i++) {
        cout << "Loop iteration: " << i << endl;
    }
    return 0;
}

While/Do-While Loop:


#include <iostream>
using namespace std;

int main() {
    int i = 0;
    do {
        cout << "Loop iteration: " << i << endl;
        i++;
    } while (i < 6);
    return 0;
}
